    <script>
        function calculate() {
            const g = document.getElementById('gender').value;
            const a = parseFloat(document.getElementById('age').value);
            const w = parseFloat(document.getElementById('weight').value);
            const h = parseFloat(document.getElementById('height').value);
            const f = parseFloat(document.getElementById('activity').value);
            
            if(!a || !w || !h) return alert("INPUT ERROR: FILL ALL FIELDS");

            // Mifflin-St Jeor Formula
            let bmr = (10 * w) + (6.25 * h) - (5 * a);
            bmr = (g === 'male') ? bmr + 5 : bmr - 161;
            
            const tdee = Math.round(bmr * f);
            
            document.getElementById('results').style.display = 'block';
            document.getElementById('maint').innerText = tdee;
            document.getElementById('loss').innerText = tdee - 500;
            document.getElementById('gain').innerText = tdee + 500;
        }
    </script>
